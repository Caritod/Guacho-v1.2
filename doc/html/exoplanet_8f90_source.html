59 subroutine init_exo()
60 
61  use constants, only : msun, yr, ggrav, pi, mjup, au, day,rjup
62  use globals, only :rank, dx
63  implicit none
64  real :: amdot ! mdot_star (MSUN/yr)
65  real :: ampdot ! mdot_planet (g/s)
66 #if defined(PMHD) || defined(MHD)
67  real :: bsw, bsc
68 #endif
69 
70  !----------------STAR PARAMETERS ------------------
71  masss = 1.1*msun
72  amdot = 2.e-14*msun/yr ! Stellar Mass Loss rate (g s^-1)
73 
74  tsw = 3.0e6 !************ ! Stellar temperature (K)
75  ! Stellar wind, imposed at the 1.5x sonic point (cm)
76  rsw = 1.5*ggrav*masss/2./(rg*tsw/0.6) !********************
77 
78  vsw = 372.e5 !************* ! Stellar wind velocity (cm/s)
79  dsw =((amdot/rsw)/(4*pi*rsw*vsw)) ! Stellar density @RS (g cm^-3)
80 
81 #if defined(PMHD) || defined(MHD)
82  bsw =1.0e-4 ! Stellar magnetic field (g)
83 #endif
84 
85  !----------------PLANET PARAMETERS------------------
86  massp =0.67*mjup
87 
88  ampdot=2.e10 !*********** ! Planetary Mass Loss rate (g/s)
89 
90  tpw = 1.e4 ! Planets temperature
91  rpw =3.*1.38*rjup ! Planetary wind radius (cm)
92  vpw = 10.e5 ! Planets wind velocity (cm/s)
93  dpw=((ampdot/rpw)/(4*pi*rpw*vpw)) ! Planetary wind density
94 
95 
96  !ORBITAL PARAMETERS
97  rorb=.047*au!0.47**AU
98  torb=3.52*day
99 
100  ! change to code units
101  dsw=dsw/rhosc
102  vsw=vsw/sqrt(vsc2)
103  tsw=tsw/tempsc
104  rsw=rsw/rsc
105 
106 #if defined(PMHD) || defined(MHD)
107  bsw=bsw/bsc
108 #endif
109 
110  dpw=dpw/rhosc
111  vpw=vpw/sqrt(vsc2)
112  tpw=tpw/tempsc
113  rpw=rpw/rsc
114 
115  rorb=rorb/rsc
116  torb=torb/tsc
117 
118  ! initial position
119  xp=rorb*cos(-25.*pi/180.)
120  yp=0.
121  zp=rorb*sin(-25.*pi/180.)
122 
123 end subroutine init_exo

134 subroutine impose_exo(u,time)
135 
136  use constants, only : pi
137  use globals, only : coords, dx, dy, dz, rank
138  implicit none
139  real, intent(out) :: u(neq,nxmin:nxmax,nymin:nymax,nzmin:nzmax)
140  real, intent (in) :: time
141  real :: x, y, z, xpl, ypl, zpl
142  real :: velx, vely, velz, rads, dens, radp, phi
143  real :: vxorb, vyorb, vzorb, omega ! cpi
144  integer :: i,j,k
145 
146  phi=-25.*pi/180.
147 
148  omega=2.*pi/torb
149 
150  xp=rorb*cos(omega*time+phi)
151  zp=rorb*sin(omega*time+phi)
152 
153  !Orbital planetary velocity (moves in the xz-plane)
154  vxorb=-omega*rorb*sin(omega*time+phi)
155  vzorb= omega*rorb*cos(omega*time+phi)
156  vyorb=0.
157 
158  do i=nxmin,nxmax
159  do j=nymin,nymax
160  do k=nzmin,nzmax
161 
162  ! Position measured from the centre of the grid (star)
163  x=(float(i+coords(0)*nx-nxtot/2)+0.5)*dx
164  y=(float(j+coords(1)*ny-nytot/2)+0.5)*dy
165  z=(float(k+coords(2)*nz-nztot/2)+0.5)*dz
166 
167  ! Position measured from the centre of the planet
168  xpl=x-xp
169  ypl=y
170  zpl=z-zp
171 
172 #if defined(PMHD) || defined(MHD)
173  cpi=b0*(rsw/(rads+1.e-30))**3/(2.*(rads+1.e-30)**2)
174 #endif
175  ! Distance from the centre of the star
176  rads=sqrt(x**2+y**2+z**2)
177 
178  ! Distance from the centre of the planet
179  radp=sqrt(xpl**2+ypl**2+zpl**2)
180 
181  ! IF INSIDE THE STAR
182  if( rads <= rsw) then
183  if(rads == 0.) rads=dx*0.10
184 
185  velx=vsw*x/rads
186  vely=vsw*y/rads
187  velz=vsw*z/rads
188  dens=dsw*rsw**2/rads**2
189  ! total density and momena
190  u(1,i,j,k) = dens
191  u(2,i,j,k) = dens*velx
192  u(3,i,j,k) = dens*vely
193  u(4,i,j,k) = dens*velz
194  ! magnetic field
195 #if defined(PMHD) || defined(MHD)
196  u(6,i,j,k) = 3.*y*x*cpi
197  u(7,i,j,k) = (3.*y**2-rads**2)*cpi
198  u(8,i,j,k) = 3.*y*z*cpi
199 #endif
200 
201 #ifdef MHD
202  ! total energy
203  u(5,i,j,k)=0.5*dens*vsw**2 &
204  + cv*dens*2.*1.9999*tsw &
205  + 0.5*(u(6,i,j,k)**2+u(7,i,j,k)**2+u(8,i,j,k)**2)
206 #else
207  ! total energy
208  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
209  + cv*dens*1.9999*tsw
210 #endif
211 
212 #ifdef PASSIVES
213  ! density of neutrals
214  u(neqdyn+1,i,j,k)= 0.0001*dens
215  ! passive scalar (h-hot, c-cold, i-ionized, n-neutral)
216  u(neqdyn+2,i,j,k)= dens ! passive scalar
217 
218 ! u(neqdyn+3,i,j,k)= 0.00001*dens ! xci*rho
219 ! u(neqdyn+4,i,j,k)= 0.00001*dens ! xhn*rho
220 ! u(neqdyn+5,i,j,k)= 0.00001*dens ! xcn*rho
221 
222 
223 #endif
224  ! IF INSIDE THE PLANET
225  else if(radp <= rpw) then
226 
227  if(radp == 0.) radp=dx*0.10
228 
229  velx=vxorb+vpw*xpl/radp
230  vely=vyorb+vpw*ypl/radp
231  velz=vzorb+vpw*zpl/radp
232  dens=dpw*rpw**2/radp**2
233  ! total density and momenta
234  u(1,i,j,k) = dens
235  u(2,i,j,k) = dens*velx
236  u(3,i,j,k) = dens*vely
237  u(4,i,j,k) = dens*velz
238  ! Magnetic fields (IF MHD or PMHD)
239 #if defined(PMHD) || defined(MHD)
240  u(6,i,j,k) = 0.0
241  u(7,i,j,k) = 0.0
242  u(8,i,j,k) = 0.0
243 #endif
244  ! energy
245 #ifdef MHD
246  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
247  + cv*dens* 1.0001*tpw &
248  + 0.5*(u(6,i,j,k)**2+u(7,i,j,k)**2+u(8,i,j,k)**2)
249 #else
250  u(5,i,j,k)=0.5*dens*(velx**2+vely**2+velz**2) &
251  + cv*dens*1.0001*tpw
252 #endif
253 
254 #ifdef PASSIVES
255  ! density of neutrals
256  u(neqdyn+1,i,j,k)=0.9999*dens
257  ! passive scalar (h-hot, c-cold, i-ionized, n-neutral)
258  u(neqdyn+2,i,j,k)= -dens ! passive scalar
259 
260 ! u(neqdyn+3,i,j,k)= 0.00001*dens ! xci*rho
261 ! u(neqdyn+4,i,j,k)= 0.00001*dens ! xhn*rho
262 ! u(neqdyn+5,i,j,k)= 0.99999*dens ! xcn*rho
263 #endif
264  end if
265 
266  end do
267  end do
268  end do
269 
270 end subroutine impose_exo
