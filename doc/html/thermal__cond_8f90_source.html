61 subroutine dt_cond(dt)
62  implicit none
63  real, intent(out) :: dt
64  real :: dtp, cs, ddx
65  integer :: i, j, k, err
66  !
67  dtp=1.e30
68  ddx=min(dx ,dy)
69  ddx=min(ddx,dz)
70  !
71  do i=1,nx
72  do j=1,ny
73  do k=1,nz
74  !
75  call csound(primit(5,i,j,k),primit(1,i,j,k),cs)
76  ! spitzer timescale
77  dtp=min( dtp, cv*primit(5,i,j,k)*psc*(ddx*rsc)**2/(ksp(temp(i,j,k))*temp(i,j,k) ) )

80  !
81  end do

83  end do

142 subroutine heatfluxes()
143  use sound
144  implicit none
145  integer :: i, j, k
146  real, parameter :: clight=3.e10, phi=0.3
147  real:: cs, coef, dTx, dTy, dTz, meanT, meanP, meanDens, yhp
148 
149  do i=0,nx
150  do j=0,ny
151  do k=0,nz
152 
153  yhp=1.-primit(neqdyn+1,i,j,k)/primit(1,i,j,k)
154 
155  ! get the flux in the X direction
156  if (temp(i,j,k) == temp(i+1,j,k) ) then
157  f(5,i,j,k)=0.
158  else
159  meanp = 0.5*(primit(5,i,j,k)+primit(5,i+1,j,k))
160  meandens= 0.5*(primit(1,i,j,k)+primit(1,i+1,j,k))
161  meant = 0.5*( temp(i,j,k)+ temp(i+1,j,k))
162  call csound(meanp,meandens,cs)
163  cs=min(cs*sqrt(vsc2),clight)
164 
165  dtx=(temp(i+1,j,k)-temp(i,j,k))/(dx*rsc)
166  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dtx) )
167 
168  f(5,i,j,k)=-coef*dtx*yhp
169 
170  end if
171  ! get the flux in the Y direction
172  if (temp(i,j,k) == temp(i,j+1,k) ) then
173  g(5,i,j,k)=0.
174  else
175  meanp = 0.5*(primit(5,i,j,k)+primit(5,i,j+1,k))
176  meandens= 0.5*(primit(1,i,j,k)+primit(1,i,j+1,k))
177  meant = 0.5*( temp(i,j,k)+ temp(i,j+1,k))
178  call csound(meanp,meandens,cs)
179  cs=min(cs*sqrt(vsc2),clight)
180 
181  dty=(temp(i,j+1,k)-temp(i,j,k))/(dy*rsc)
182  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dty) )
183  g(5,i,j,k)=-coef*dty*yhp
184 
185  end if
186  ! get the flux in the Z direction
187  if (temp(i,j,k) == temp(i,j,k+1) ) then
188  h(5,i,j,k)=0.
189  else
190  meanp = 0.5*(primit(5,i,j,k)+primit(5,i,j,k+1))
191  meandens= 0.5*(primit(1,i,j,k)+primit(1,i,j,k+1))
192  meant = 0.5*( temp(i,j,k) + temp(i,j,k+1))
193  call csound(meanp,meandens,cs)
194  cs=min(cs*sqrt(vsc2),clight)
195 
196  dtz=(temp(i,j,k+1)-temp(i,j,k))/(dz*rsc)
197  coef=min( ksp(meant) , 5.*ph*cs*meanp*psc/abs(dtz) )
198 
199  h(5,i,j,k)=-coef*dtz*yhp
200 
201  end if
202 
203  end do
204  end do
205  end do
206  !
207 end subroutine heatfluxes

407 subroutine thermal_conduction(dt)
408 
409  implicit none
410  real, intent(in) :: dt
411  real :: dtcond, dts,fstep
412  integer :: n,i,j,k, nsteps
413  logical :: SuperStep
414  ! get the conduction timescale
415  call dt_cond(dtcond)
416 
417 !!$ if (dtcond < dt) then
418 !!$ SuperStep=.True.
419 !!$ call ST_steps(dt/dtcond,Nsteps,fstep)
420 !!$ else
421 !!$ SuperStep=.False.
422 !!$ fstep=1.
423 !!$ nsteps=1
424 !!$ end if
425 
426  !Superstepping off
427  fstep=min(dt/dtcond+1.,100.)
428  nsteps=int(fstep)
429  ! here i take care of the transformation from cgs to code units
430  dts=dt/float(nsteps)/psc/rsc
431 
432  if (rank == master) print'(a,es10.3,a,i4,a)',"dt/dtcond= ", dt/dtcond, " => ", nsteps," Ssteps"
433 
434  steps : do n=1,nsteps
435 
436 !!$ if (SuperStep) then
437 !!$ dts=dt*fstep*substep(n,Nsteps,nu)/Psc/rsc
438 !!$ else
439 !!$ dts=dt/Psc/rsc
440 !!$ end if
441 
442  ! show progress bar (comment to run in batch)
443  !if (rank == master )call progress(n,nsteps)
444 
445  ! get the heat fluxes
446  call heatfluxes()
447  ! update the conserved and primitives
448  do i=1,nx
449  do j=1,ny
450  do k=1 ,nz
451  u(5,i,j,k)=u(5,i,j,k)-dts*( ( f(5,i,j,k) - f(5,i-1,j,k) )/dx &
452  +( g(5,i,j,k) - g(5,i,j-1,k) )/dy &
453  +( h(5,i,j,k) - h(5,i,j,k-1) )/dz )
454  end do
455  end do
456  end do
457 
458  ! boundary conditions
459  ! (only one layer of u(5,:,:,:) is exchanged )
460  call thermal_bounds()
461  ! update primitives and Temperature
462  call calcprim(u, primit)
463 
464  end do steps
465 
466 end subroutine thermal_conduction
